Example 1: Create History Manager

Initialize history tracking

// Create history manager
const history = createHistoryManager(initialState);
// Can optionally set max history size
const history = createHistoryManager([], 50);

Example 2: Push State

Add state to history

// Push new state
const success = pushState(history, newNotes);
// Current state is now newNotes
// Undo/redo becomes available

Example 3: Undo & Redo

Navigate through history

// Check if can undo
if (canUndo(history)) {
  const previous = undo(history);
}

// After undo, check for redo
if (canRedo(history)) {
  const next = redo(history);
}

Example 4: Get Current State

Retrieve current state

// Get current state at history position
const current = getCurrentState(history);
// Use current for rendering UI

Example 5: Get Full History

View all historical states

// Get all states from start to current
const allStates = getHistory(history);
// Returns: [initial, state1, state2, current]
// Does not include redo states

Example 6: History Statistics

Analyze history state

// Get stats
const stats = getHistoryStats(history);
// {
//   currentIndex: 2,
//   totalStates: 4,
//   undoCount: 2,
//   redoCount: 1
// }

Example 7: Check Capabilities

Know when undo/redo available

// Check before undo/redo
if (canUndo(history)) {
  undoButton.disabled = false;
} else {
  undoButton.disabled = true;
}

if (canRedo(history)) {
  redoButton.disabled = false;
}

Example 8: Clear History

Reset history state

// Clear all history
clearHistory(history);
// Now at initial state
// No undo or redo available
